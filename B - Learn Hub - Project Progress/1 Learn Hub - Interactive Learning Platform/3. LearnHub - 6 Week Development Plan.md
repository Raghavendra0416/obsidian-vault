async function binarySearch(arr, target) {
  let left = 0;
  let right = arr.length - 1;
  
  while(left <= right) {
    const mid = Math.floor((left + right) / 2);
    highlightBars(left, mid, right);
    await sleep(1000);
    
    if(arr[mid] === target) {
      return mid; // Found!
    } else if(arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }
  return -1; // Not found
}

function calculateProgress(course) {
  const total = course.modules.length;
  const completed = course.modules.filter(m => m.isCompleted).length;
  return Math.round((completed / total) * 100);
}

function saveProgress(userId, courseId, moduleId) {
  const progress = JSON.parse(localStorage.getItem('userProgress')) || {};
  if(!progress[userId]) progress[userId] = {};
  if(!progress[userId][courseId]) progress[userId][courseId] = [];
  
  progress[userId][courseId].push(moduleId);
  localStorage.setItem('userProgress', JSON.stringify(progress));
}

  let timeoutId;
  return function(...args) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => func.apply(this, args), delay);
  };
}
